Header <boost/text/case_mapping.hpp>

namespace boost {
  namespace text {
    struct next_word_break_callable;

    enum case_language;
    template<typename CPIter, typename Sentinel> 
      bool is_lower(CPIter, Sentinel);
    template<typename CPRange> bool is_lower(CPRange &&);
    template<typename GraphemeRange> bool is_lower(GraphemeRange &&);
    template<typename CPIter, typename Sentinel, typename OutIter> 
      OutIter to_lower(CPIter, CPIter, Sentinel, OutIter, 
                       case_language = case_language::other);
    template<typename CPRange, typename OutIter> 
      OutIter to_lower(CPRange &&, OutIter, 
                       case_language = case_language::other);
    template<typename GraphemeRange, typename OutIter> 
      OutIter to_lower(GraphemeRange &&, OutIter, 
                       case_language = case_language::other);
    template<typename CPIter, typename Sentinel, 
             typename NextWordBreakFunc = next_word_break_callable> 
      bool is_title(CPIter, Sentinel, NextWordBreakFunc = NextWordBreakFunc{});
    template<typename CPRange, 
             typename NextWordBreakFunc = next_word_break_callable> 
      bool is_title(CPRange &&, NextWordBreakFunc = NextWordBreakFunc{});
    template<typename GraphemeRange, 
             typename NextWordBreakFunc = next_word_break_callable> 
      bool is_title(GraphemeRange &&, NextWordBreakFunc = NextWordBreakFunc{});
    template<typename CPIter, typename Sentinel, typename OutIter, 
             typename NextWordBreakFunc = next_word_break_callable> 
      OutIter to_title(CPIter, CPIter, Sentinel, OutIter, 
                       case_language = case_language::other, 
                       NextWordBreakFunc = NextWordBreakFunc{});
    template<typename CPRange, typename OutIter, 
             typename NextWordBreakFunc = next_word_break_callable> 
      OutIter to_title(CPRange &&, OutIter, 
                       case_language = case_language::other, 
                       NextWordBreakFunc = NextWordBreakFunc{});
    template<typename GraphemeRange, typename OutIter, 
             typename NextWordBreakFunc = next_word_break_callable> 
      OutIter to_title(GraphemeRange &&, OutIter, 
                       case_language = case_language::other, 
                       NextWordBreakFunc = NextWordBreakFunc{});
    template<typename CPIter, typename Sentinel> 
      bool is_upper(CPIter, Sentinel);
    template<typename CPRange> bool is_upper(CPRange &&);
    template<typename GraphemeRange> bool is_upper(GraphemeRange &&);
    template<typename CPIter, typename Sentinel, typename OutIter> 
      OutIter to_upper(CPIter, CPIter, Sentinel, OutIter, 
                       case_language = case_language::other);
    template<typename CPRange, typename OutIter> 
      OutIter to_upper(CPRange &&, OutIter, 
                       case_language = case_language::other);
    template<typename GraphemeRange, typename OutIter> 
      OutIter to_upper(GraphemeRange &&, OutIter, 
                       case_language = case_language::other);
  }
}
