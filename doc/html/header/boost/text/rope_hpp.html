Header <boost/text/rope.hpp>

namespace boost {
  namespace text {
    struct rope;
    rope::iterator begin(rope const & t);
    rope::iterator end(rope const & t);
    rope::iterator cbegin(rope const & t);
    rope::iterator cend(rope const & t);
    rope::reverse_iterator rbegin(rope const & t);
    rope::reverse_iterator rend(rope const & t);
    rope::reverse_iterator crbegin(rope const & t);
    rope::reverse_iterator crend(rope const & t);
    bool operator==(text const & lhs, rope_view rhs);
    bool operator==(rope_view lhs, text const & rhs);
    bool operator!=(text const & lhs, rope_view rhs);
    bool operator!=(rope_view lhs, text const & rhs);
    bool operator==(text const & lhs, rope rhs);
    bool operator==(rope lhs, text const & rhs);
    bool operator!=(text const & lhs, rope rhs);
    bool operator!=(rope lhs, text const & rhs);
    bool operator==(rope const & lhs, rope_view rhs);
    bool operator==(rope_view lhs, rope const & rhs);
    bool operator!=(rope const & lhs, rope_view rhs);
    bool operator!=(rope_view lhs, rope const & rhs);
    bool operator==(rope const & lhs, rope const & rhs);
    bool operator!=(rope const & lhs, rope const & rhs);
    rope operator+(rope, char const *);
    rope operator+(char const *, rope const &);
    rope operator+(rope, rope const &);
    rope operator+(rope, rope_view);
    rope operator+(rope_view, rope const &);
    template<typename CharRange> rope operator+(rope, CharRange const &);
    template<typename CharRange> 
      rope operator+(CharRange const &, rope const &);
    text & operator+=(text &, rope_view);
    rope operator+(text, rope_view);
    rope operator+(rope_view, text const &);
    text & operator+=(text &, rope const &);
    rope operator+(text, rope const &);
    rope operator+(rope, text const &);
  }
}
