Header <boost/text/unencoded_rope.hpp>

namespace boost {
  namespace text {
    namespace v1 {
      struct unencoded_rope;
      bool operator==(char const * lhs, unencoded_rope const & rhs);
      bool operator==(unencoded_rope const & lhs, char const * rhs);
      bool operator!=(char const * lhs, unencoded_rope const & rhs);
      bool operator!=(unencoded_rope const & lhs, char const * rhs);
      bool operator<(char const * lhs, unencoded_rope const & rhs);
      bool operator<(unencoded_rope const & lhs, char const * rhs);
      bool operator<=(char const * lhs, unencoded_rope const & rhs);
      bool operator<=(unencoded_rope const & lhs, char const * rhs);
      bool operator>(char const * lhs, unencoded_rope const & rhs);
      bool operator>(unencoded_rope const & lhs, char const * rhs);
      bool operator>=(char const * lhs, unencoded_rope const & rhs);
      bool operator>=(unencoded_rope const & lhs, char const * rhs);
      template<typename CharRange> 
        auto operator==(CharRange const & lhs, unencoded_rope const & rhs);
      template<typename CharRange> 
        auto operator==(unencoded_rope const & lhs, CharRange const & rhs);
      template<typename CharRange> 
        auto operator!=(CharRange const & lhs, unencoded_rope const & rhs);
      template<typename CharRange> 
        auto operator!=(unencoded_rope const & lhs, CharRange const & rhs);
      template<typename CharRange> 
        auto operator<(CharRange const & lhs, unencoded_rope const & rhs);
      template<typename CharRange> 
        auto operator<(unencoded_rope const & lhs, CharRange const & rhs);
      template<typename CharRange> 
        auto operator<=(CharRange const & lhs, unencoded_rope const & rhs);
      template<typename CharRange> 
        auto operator<=(unencoded_rope const & lhs, CharRange const & rhs);
      template<typename CharRange> 
        auto operator>(CharRange const & lhs, unencoded_rope const & rhs);
      template<typename CharRange> 
        auto operator>(unencoded_rope const & lhs, CharRange const & rhs);
      template<typename CharRange> 
        auto operator>=(CharRange const & lhs, unencoded_rope const & rhs);
      template<typename CharRange> 
        auto operator>=(unencoded_rope const & lhs, CharRange const & rhs);
      unencoded_rope operator+(unencoded_rope, char const *);
      unencoded_rope operator+(char const *, unencoded_rope);
      unencoded_rope operator+(unencoded_rope, unencoded_rope);
      unencoded_rope operator+(unencoded_rope, unencoded_rope_view);
      unencoded_rope operator+(unencoded_rope_view, unencoded_rope);
      unencoded_rope operator+(unencoded_rope, string &&);
      unencoded_rope operator+(string &&, unencoded_rope);
      template<typename CharRange> 
        unencoded_rope & operator+(unencoded_rope, CharRange const &);
      template<typename CharRange> 
        unencoded_rope & operator+(CharRange const &, unencoded_rope const &);
      unencoded_rope_view::iterator begin(unencoded_rope_view rv);
      unencoded_rope_view::iterator end(unencoded_rope_view rv);
      unencoded_rope_view::iterator cbegin(unencoded_rope_view rv);
      unencoded_rope_view::iterator cend(unencoded_rope_view rv);
      unencoded_rope_view::reverse_iterator rbegin(unencoded_rope_view rv);
      unencoded_rope_view::reverse_iterator rend(unencoded_rope_view rv);
      unencoded_rope_view::reverse_iterator crbegin(unencoded_rope_view rv);
      unencoded_rope_view::reverse_iterator crend(unencoded_rope_view rv);
      std::ostream & operator<<(std::ostream &, unencoded_rope_view);
    }
  }
}
