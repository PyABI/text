Header <boost/text/rope_view.hpp>

namespace boost {
  namespace text {
    struct rope_view;
    bool operator==(rope_view lhs, rope_view rhs);
    bool operator!=(rope_view lhs, rope_view rhs);
    rope_view::iterator begin(rope_view tv);
    rope_view::iterator end(rope_view tv);
    rope_view::iterator cbegin(rope_view tv);
    rope_view::iterator cend(rope_view tv);
    rope_view::reverse_iterator rbegin(rope_view tv);
    rope_view::reverse_iterator rend(rope_view tv);
    rope_view::reverse_iterator crbegin(rope_view tv);
    rope_view::reverse_iterator crend(rope_view tv);
  }
}
